# factordb yafu composite worker
work=25; start=0; range=1000; numcomps=1; digits=88; cd ~/yafu; while :; do rm *.dat; printf "http://factordb.com/listtype.php?t=3&mindig=$digits&perpage=$numcomps&start=$(($RANDOM%$range+$start))&download=1" | xargs -I {} curl -s '{}' | ./yafu -threads 16 -snfs_xover 90 -work $work -one -of factors.txt "factor(@)" 1> /dev/stderr; cat factors.txt | tr "/" "," | sed -nr 's/\(([0-9]+)\),(([0-9]+,?)+)/\1=[\2]/p' | jq -sRr @uri | xargs -I {} curl -s -G 'http://factordb.com/report.php?report={}&format=0' | lynx -dump -stdin -nolist -width=$(tput cols); rm factors.txt; done;
